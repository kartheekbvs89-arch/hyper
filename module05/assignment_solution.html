Task 1 — SmallCNN

sol_task1_small_cnn.py
# ─── sol_task1_small_cnn.py ────────────────────────────────────────────────────
import torch
import torch.nn as nn

class SmallCNN(nn.Module):
    """Two conv blocks (Conv→BN→ReLU→MaxPool) + two FC layers."""

    def __init__(self, n_filters: int = 32, dropout: float = 0.0,
                 num_classes: int = 10):
        super().__init__()
        f1, f2 = n_filters, 2 * n_filters

        self.conv_block_1 = nn.Sequential(
            nn.Conv2d(3, f1, kernel_size=3, padding=1, bias=False),
            nn.BatchNorm2d(f1),
            nn.ReLU(inplace=True),
            nn.MaxPool2d(kernel_size=2, stride=2),       # 32x32 → 16x16
        )
        self.conv_block_2 = nn.Sequential(
            nn.Conv2d(f1, f2, kernel_size=3, padding=1, bias=False),
            nn.BatchNorm2d(f2),
            nn.ReLU(inplace=True),
            nn.MaxPool2d(kernel_size=2, stride=2),       # 16x16 → 8x8
        )
        self.classifier = nn.Sequential(
            nn.Flatten(),
            nn.Linear(f2 * 8 * 8, 128),
            nn.ReLU(inplace=True),
            nn.Dropout(p=dropout),
            nn.Linear(128, num_classes),
        )

    def forward(self, x):
        x = self.conv_block_1(x)
        x = self.conv_block_2(x)
        return self.classifier(x)

# ─── Sanity check ────────────────────────────────────────────────────────────
model = SmallCNN(n_filters=32, dropout=0.1)
x = torch.randn(2, 3, 32, 32)
out = model(x)
assert out.shape == (2, 10), f"Expected (2, 10), got {out.shape}"
n_params = sum(p.numel() for p in model.parameters())
print(f"OK — output shape {out.shape}, params {n_params:,}")
# With n_filters=32: ~585k params. With n_filters=128: ~8.5M params.

Note that the Conv2d layers use bias=False — this is standard when followed by BatchNorm2d, because BN's learned \beta parameter already plays the role of a bias and a separate conv bias would be redundant (and would just get normalized away). This is one of those small details that distinguishes a careful implementation from a naive one.

Task 2 — Optuna Objective

sol_task2_objective.py
# ─── sol_task2_objective.py ───────────────────────────────────────────────────
import optuna
import torch
import torch.nn as nn
import torch.optim as optim
from torch.utils.data import DataLoader

device = "cuda" if torch.cuda.is_available() else "cpu"
EPOCHS_PER_TRIAL = 5

def build_optimizer(model, name, lr, wd):
    """Apply weight decay ONLY to weight matrices — not biases or BN params."""
    decay, no_decay = [], []
    for m in model.modules():
        if isinstance(m, nn.Linear):
            decay.append(m.weight)
            if m.bias is not None:
                no_decay.append(m.bias)
        elif isinstance(m, nn.Conv2d):
            decay.append(m.weight)
            if m.bias is not None:
                no_decay.append(m.bias)
        elif isinstance(m, (nn.BatchNorm2d, nn.LayerNorm)):
            if m.weight is not None:
                no_decay.append(m.weight)
            if m.bias is not None:
                no_decay.append(m.bias)
    params = [
        {"params": decay, "weight_decay": wd},
        {"params": no_decay, "weight_decay": 0.0},
    ]
    if name == "adam":
        return optim.Adam(params, lr=lr)
    if name == "adamw":
        return optim.AdamW(params, lr=lr)
    if name == "sgd_momentum":
        return optim.SGD(params, lr=lr, momentum=0.9)
    raise ValueError(name)

def objective(trial: optuna.Trial) -> float:
    # ─── 1. Suggest hyperparameters ──────────────────────────────────────────
    lr           = trial.suggest_float("lr", 1e-5, 1e-1, log=True)
    batch_size   = trial.suggest_categorical("batch_size", [32, 64, 128, 256])
    dropout      = trial.suggest_float("dropout", 0.0, 0.5)
    n_filters    = trial.suggest_int("n_filters", 32, 128, step=16)
    weight_decay = trial.suggest_float("weight_decay", 1e-6, 1e-2, log=True)
    opt_name     = trial.suggest_categorical(
        "optimizer", ["adam", "sgd_momentum", "adamw"]
    )

    # ─── 2. Build model, optimizer, dataloaders ──────────────────────────────
    model = SmallCNN(n_filters=n_filters, dropout=dropout).to(device)
    optimizer = build_optimizer(model, opt_name, lr, weight_decay)
    criterion = nn.CrossEntropyLoss()

    train_loader = DataLoader(train_ds, batch_size=batch_size, shuffle=True,
                              num_workers=2, pin_memory=True)
    val_loader   = DataLoader(val_ds, batch_size=512, shuffle=False,
                              num_workers=2, pin_memory=True)

    # ─── 3. Train + validate, with pruning ───────────────────────────────────
    for epoch in range(EPOCHS_PER_TRIAL):
        model.train()
        for x, y in train_loader:
            x, y = x.to(device), y.to(device)
            optimizer.zero_grad()
            loss = criterion(model(x), y)
            loss.backward()
            optimizer.step()

        # Validate after each epoch.
        model.eval()
        correct, total = 0, 0
        with torch.no_grad():
            for x, y in val_loader:
                x, y = x.to(device), y.to(device)
                correct += (model(x).argmax(1) == y).sum().item()
                total += y.size(0)
        val_acc = correct / total

        # Report to Optuna for pruning decisions.
        trial.report(val_acc, epoch)
        if trial.should_prune():
            raise optuna.TrialPruned()

    return val_acc

The build_optimizer function carefully splits parameters into decay and no-decay groups. nn.Linear.weight and nn.Conv2d.weight get weight decay; biases and BatchNorm .weight/.bias do not. This is the standard pattern from the HuggingFace Transformers codebase and from most modern PyTorch recipes.

Task 3 — Run 50 Trials with MedianPruner

sol_task3_run_study.py
# ─── sol_task3_run_study.py ───────────────────────────────────────────────────
import time
import optuna
from collections import Counter

optuna.logging.set_verbosity(optuna.logging.WARNING)

study = optuna.create_study(
    direction="maximize",
    pruner=optuna.pruners.MedianPruner(
        n_startup_trials=5,   # don't prune the first 5 trials
        n_warmup_steps=1,     # don't prune before epoch 1
    ),
)

t0 = time.time()
study.optimize(objective, n_trials=50)
elapsed = time.time() - t0

print(f"Best val acc: {study.best_value:.4f}")
print(f"Best params: {study.best_params}")
print(f"Elapsed: {elapsed/60:.1f} min")

states = Counter(t.state for t in study.trials)
print(f"Trial states: {dict(states)}")

# Typical output on a single GPU:
#   Best val acc: 0.78-0.85
#   Best params: lr ~1e-3, batch_size=128, dropout~0.2-0.3,
#                n_filters~64-96, weight_decay~1e-4, optimizer='adamw'
#   Elapsed: 50-90 min
#   Trial states: ~30-40 COMPLETE, ~10-20 PRUNED

The MedianPruner keeps a running median of objective values at each step. After n_startup_trials trials and n_warmup_steps epochs, any trial whose intermediate value is worse than the median gets pruned. This typically saves 30–50% of compute on a 50-trial study with 5 epochs each.

Task 6 (Bonus) — Linear Warmup + Cosine Decay

sol_task6_warmup_cosine.py
# ─── sol_task6_warmup_cosine.py ────────────────────────────────────────────────
import numpy as np
import torch
from torch.optim.lr_scheduler import LambdaLR

def warmup_cosine_scheduler(optimizer, n_warmup: int = 3, n_total: int = 30):
    """Linear warmup for n_warmup epochs, then cosine decay to 0."""

    def lr_lambda(epoch: int) -> float:
        if epoch < n_warmup:
            # Linear warmup from 0 to peak_lr.
            return (epoch + 1) / n_warmup
        # Cosine decay from peak_lr to 0 over the remaining epochs.
        progress = (epoch - n_warmup) / max(1, (n_total - n_warmup))
        return 0.5 * (1.0 + np.cos(np.pi * progress))

    return LambdaLR(optimizer, lr_lambda)

# ─── Verify the LR curve ─────────────────────────────────────────────────────
import matplotlib.pyplot as plt

dummy_opt = torch.optim.SGD([torch.nn.Parameter(torch.zeros(1))], lr=1.0)
sched = warmup_cosine_scheduler(dummy_opt, n_warmup=3, n_total=30)
lrs = []
for _ in range(30):
    lrs.append(dummy_opt.param_groups[0]["lr"])
    sched.step()

Warmup+cosine typically wins by 0.3–1.0 percentage points on SmallCNN/CIFAR-10. The win is small here because the model is small, the LR is moderate, and Adam-family optimizers already adapt their step size. The pattern is sharper on larger models (ResNet-50, Transformers) and higher peak LRs, where the early-training instability that warmup prevents is more pronounced.

The reason warmup helps at all on this setup is subtle: even Adam benefits from a few epochs of small-step training to get its second-moment estimate v_t populated with reliable values. Once v_t is stable, the full LR can be applied safely. Without warmup, the first few epochs of Adam training have noisy v_t and the effective step size lr / sqrt(v_t) is occasionally way too large for some parameters. Warmup damps this.
