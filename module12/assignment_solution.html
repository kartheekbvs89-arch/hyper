Task 1 — Dual-Objective Function

snippet.py
# ─── sol_task1_dual_objective.py ────────────────────────────────────────────────
import time
import numpy as np
from xgboost import XGBClassifier
from sklearn.datasets import load_breast_cancer
from sklearn.model_selection import StratifiedKFold, cross_val_score

X, y = load_breast_cancer(return_X_y=True)
cv = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)


def measure_latency(model, X, n_repeats=5, n_samples=1000):
    """Median per-sample inference latency in milliseconds.

    Warms up the model with 10 calls, then times 5 repeats of 100 predict()
    calls on a random subsample of size n_samples.
    """
    rng = np.random.default_rng(42)
    idx = rng.choice(len(X), size=min(n_samples, len(X)), replace=True)
    Xs = X[idx]

    # Warm up
    for _ in range(10):
        model.predict(Xs)

    # Measure
    times = []
    for _ in range(n_repeats):
        t0 = time.perf_counter()
        for _ in range(100):
            model.predict(Xs)
        t1 = time.perf_counter()
        # Convert to ms per sample
        times.append((t1 - t0) / 100 / len(Xs) * 1000)
    return float(np.median(times))


def objective(trial):
    """Returns (accuracy, latency_ms) for a single XGBoost configuration.

    n_jobs is forced to 1 so latency measurements are comparable across trials.
    """
    params = {
        "learning_rate":     trial.suggest_float("learning_rate", 1e-3, 1.0, log=True),
        "max_depth":         trial.suggest_int("max_depth", 3, 10),
        "n_estimators":      trial.suggest_int("n_estimators", 50, 500),
        "subsample":         trial.suggest_float("subsample", 0.5, 1.0),
        "colsample_bytree":  trial.suggest_float("colsample_bytree", 0.5, 1.0),
        "min_child_weight":  trial.suggest_int("min_child_weight", 1, 10),
        "eval_metric":       "logloss",
        "tree_method":       "hist",
        "n_jobs":            1,           # critical: single-thread for fair latency
        "random_state":      42,
    }
    model = XGBClassifier(**params)
    scores = cross_val_score(model, X, y, cv=cv, scoring="accuracy")
    acc = float(scores.mean())

    # Fit on all data, then measure latency
    model.fit(X, y)
    lat = measure_latency(model, X)

    return acc, lat


# Sanity check
import optuna
test_study = optuna.create_study(directions=["maximize", "minimize"])
test_study.optimize(objective, n_trials=3)
for t in test_study.trials:
    print(f"Trial {t.number}: acc={t.values[0]:.4f}  lat={t.values[1]:.5f} ms")

Sanity-check output: each trial should report acc in [0.94, 0.99] and lat in [0.001, 0.05] ms. Latency varies because larger n_estimators and deeper max_depth produce larger trees that take longer to traverse during inference.

Task 2 — NSGA-II Multi-Objective Study

snippet.py
# ─── sol_task2_nsgaii_study.py ──────────────────────────────────────────────────
import optuna
import time

t0 = time.perf_counter()

study = optuna.create_study(
    directions=["maximize", "minimize"],
    sampler=optuna.samplers.NSGAIISampler(seed=42),
    storage="sqlite:///assign12_multiobj.db",
    study_name="assign12_multiobj",
    load_if_exists=True,
)
study.optimize(objective, n_trials=100)

elapsed = time.perf_counter() - t0
print(f"\n100 trials completed in {elapsed:.1f}s")

pareto = study.best_trials
print(f"Pareto front: {len(pareto)} trials")
for t in sorted(pareto, key=lambda t: -t.values[0])[:8]:
    print(f"  trial {t.number:3d}  acc={t.values[0]:.4f}  lat={t.values[1]:.5f} ms")

Typical output: ~80 seconds for 100 trials. Pareto front has 18–25 trials. Top-8 by accuracy all have acc > 0.975 and lat < 0.02 ms.

Task 3 — Pareto Front Plot

snippet.py
# ─── sol_task3_pareto_plot.py ───────────────────────────────────────────────────
import matplotlib.pyplot as plt
import numpy as np

pareto_nums = {t.number for t in study.best_trials}

trials = [t for t in study.trials if t.state == optuna.trial.TrialState.COMPLETE]
nums   = np.array([t.number for t in trials])
accs   = np.array([t.values[0] for t in trials])
lats   = np.array([t.values[1] for t in trials])
is_par = np.array([t.number in pareto_nums for t in trials])

fig, ax = plt.subplots(figsize=(9, 5.5), constrained_layout=True)
# Dominated trials (gray, small)
ax.scatter(lats[~is_par], accs[~is_par], c=nums[~is_par], cmap="viridis",
           s=30, alpha=0.5, marker="o", label="dominated")
# Pareto-optimal trials (green, large, white edge)
sc = ax.scatter(lats[is_par], accs[is_par], c=nums[is_par], cmap="viridis",
                s=120, marker="o", edgecolors="white", linewidth=1.5,
                label="Pareto-optimal")
# Connect the Pareto front with a line
order = np.argsort(lats[is_par])
ax.plot(lats[is_par][order], accs[is_par][order], "-",
        color="#34d399", linewidth=2, alpha=0.7, label="Pareto front")

ax.set_xlabel("Inference latency (ms per sample)")
ax.set_ylabel("Validation accuracy")
ax.set_title("Multi-objective Pareto front: XGBoost on breast_cancer")
cbar = fig.colorbar(sc, ax=ax)
cbar.set_label("Trial number")
ax.legend(loc="lower right")
plt.savefig("assign12_task3_pareto.png", dpi=120)
plt.show()

The Pareto front appears as a green curve in the upper-left of the scatter plot — high accuracy, low latency. Late trials (high trial numbers, light green on the colormap) cluster near the front; early trials (low numbers, dark purple) are scattered. This is the visual signature of NSGA-II converging toward the front as the search progresses.

Task 4 — Two Operating Points

snippet.py
# ─── sol_task4_operating_points.py ─────────────────────────────────────────────
pareto = study.best_trials

# Point A: highest accuracy subject to latency < 1 ms
feasible_A = [t for t in pareto if t.values[1] < 1.0]
point_A = max(feasible_A, key=lambda t: t.values[0])

# Point B: lowest latency subject to accuracy > 0.97
feasible_B = [t for t in pareto if t.values[0] > 0.97]
point_B = min(feasible_B, key=lambda t: t.values[1])

print("Operating Point A (max accuracy, latency < 1ms):")
print(f"  Trial number:    {point_A.number}")
print(f"  Accuracy:        {point_A.values[0]:.4f}")
print(f"  Latency (ms):    {point_A.values[1]:.5f}")
print(f"  Hyperparameters: {point_A.params}")

print("\nOperating Point B (min latency, accuracy > 0.97):")
print(f"  Trial number:    {point_B.number}")
print(f"  Accuracy:        {point_B.values[0]:.4f}")
print(f"  Latency (ms):    {point_B.values[1]:.5f}")
print(f"  Hyperparameters: {point_B.params}")

Discussion

(a) Real-time ad-bidding system: I would deploy Point B. In ad bidding, every millisecond of latency costs money — the bid must arrive before the auction closes, which is typically a 10–100ms window including network round-trip. The 0.5–1% accuracy gap between A and B is worth far less than the latency savings: a 2× faster model means you can bid on 2× more auctions, which more than compensates for the small accuracy hit.

(b) Medical screening tool: I would deploy Point A. In medical screening, false negatives (missed cancer) have catastrophic cost — both clinically and legally. The latency budget is much more generous (the patient waited days for the appointment; they can wait another 50ms for the prediction). Maximizing accuracy subject to the SLA is the right call. If anything, I would push the latency SLA higher (5ms? 10ms?) to see whether accuracy could be pushed further.

Task 5 — Log Every Trial to MLflow

snippet.py
# ─── sol_task5_mlflow.py ──────────────────────────────────────────────────────
import mlflow
from optuna.integration.mlflow import MLflowCallback
import optuna

mlflow.set_tracking_uri("file:./mlruns")
mlflow.set_experiment("assign12_multiobj")

callback = MLflowCallback(
    tracking_uri=mlflow.get_tracking_uri(),
    metric_name=["cv_accuracy", "latency_ms"],
)

@callback.track_in_mlflow()
def objective_mlflow(trial):
    acc, lat = objective(trial)   # reuse Task 1's objective
    return acc, lat

study_ml = optuna.create_study(
    directions=["maximize", "minimize"],
    sampler=optuna.samplers.NSGAIISampler(seed=42),
    storage="sqlite:///assign12_mlflow.db",
    study_name="assign12_mlflow",
)
study_ml.optimize(objective_mlflow, n_trials=100, callbacks=[callback])

# Find trial 50 in MLflow
client = mlflow.tracking.MlflowClient()
exp = client.get_experiment_by_name("assign12_multiobj")
runs = client.search_runs(experiment_ids=[exp.experiment_id])
trial_50 = next(
    (r for r in runs if r.data.tags.get("trial_number") == "50"),
    None,
)
if trial_50:
    print("MLflow run for trial 50:")
    print(f"  Run ID:        {trial_50.info.run_id}")
    print(f"  Params:        {trial_50.data.params}")
    print(f"  Metrics:       {trial_50.data.metrics}")
else:
    print("Trial 50 not found in MLflow. Check the tag name.")

Open mlflow ui in a terminal and navigate to http://localhost:5000. You will see 100 runs in the assign12_multiobj experiment. Filter the runs table by typing tags.trial_number = "50" in the search box to find trial 50 specifically.

Task 6 — Ray Tune Comparison

snippet.py
# ─── sol_task6_ray_tune.py ────────────────────────────────────────────────────
import os
import time
import ray
from ray import tune
from ray.tune.search.optuna import OptunaSearch
from ray.tune.schedulers import ASHAScheduler

ray.init(ignore_reinit_error=True)
X_ref, y_ref = ray.put(X), ray.put(y)


def trainable(config):
    X = ray.get(X_ref)
    y = ray.get(y_ref)
    cv = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)

    model = XGBClassifier(
        learning_rate=config["learning_rate"],
        max_depth=int(config["max_depth"]),
        n_estimators=int(config["n_estimators"]),
        subsample=config["subsample"],
        colsample_bytree=config["colsample_bytree"],
        min_child_weight=int(config["min_child_weight"]),
        eval_metric="logloss", tree_method="hist",
        n_jobs=1, random_state=42,
    )
    scores = cross_val_score(model, X, y, cv=cv, scoring="accuracy")
    acc = float(scores.mean())
    model.fit(X, y)
    lat = measure_latency(model, X)
    tune.report(cv_accuracy=acc, latency_ms=lat)


search_space = {
    "learning_rate":     tune.loguniform(1e-3, 1.0),
    "max_depth":         tune.randint(3, 11),
    "n_estimators":      tune.randint(50, 501),
    "subsample":         tune.uniform(0.5, 1.0),
    "colsample_bytree":  tune.uniform(0.5, 1.0),
    "min_child_weight":  tune.randint(1, 11),
}

# Optuna multi-objective inside Ray Tune
search_alg = OptunaSearch(
    metric=["cv_accuracy", "latency_ms"],
    mode=["max", "min"],
)

t0 = time.perf_counter()
tuner = tune.Tuner(
    trainable,
    param_space=search_space,
    tune_config=tune.TuneConfig(
        metric=["cv_accuracy", "latency_ms"],
        mode=["max", "min"],
        search_alg=search_alg,
        num_samples=100,
        max_concurrent_trials=2,
    ),
    run_config=ray.train.RunConfig(
        name="assign12_ray", storage_path="/tmp/ray_results",
    ),
)
results = tuner.fit()
ray_elapsed = time.perf_counter() - t0

print(f"Ray Tune (2 workers): {ray_elapsed:.1f}s for 100 trials")
print(f"Single-machine Optuna (from Task 2): {elapsed:.1f}s")
print(f"Speedup: {elapsed / ray_elapsed:.2f}x")

# Best result on the Pareto front
best = results.get_best_result(metric="cv_accuracy", mode="max")
print(f"\nBest by accuracy:  acc={best.metrics['cv_accuracy']:.4f}  "
      f"lat={best.metrics['latency_ms']:.5f} ms")

Typical output: Ray Tune with 2 workers completes 100 trials in ~50–60 seconds, vs ~80–100 seconds for single-machine Optuna — a ~1.5× speedup. The speedup is sublinear because (a) Ray adds scheduling overhead (~5–10%), and (b) on a 4-core laptop, 2 workers each pinning 1 thread for the latency measurement leaves fewer cores for cross-validation. On a real cluster with 8+ workers, the speedup approaches linear.

🎯 What This Assignment Teaches

  • Multi-objective optimization with NSGA-II is the production-grade answer to "I also care about latency."
  • A 100-trial NSGA-II study produces 15–25 Pareto-optimal trials — enough to make an informed product decision.
  • The choice of operating point on the Pareto front is a product decision, not a technical one. The engineer delivers the front; the product team picks the point.
  • MLflow integration adds 2 lines of code and gives you queryable per-trial logs that survive across sessions.
  • Ray Tune gives a ~1.5× speedup with 2 workers on a laptop, and approaches linear speedup on a real cluster.
  • Latency measurement is sensitive to thread count — pin n_jobs=1 and warm up the model before timing.
