sol_setup.py
# ─── sol_setup.py ─────────────────────────────────────────────────────────────
# Common setup for all task solutions. Loads the dataset, splits into
# train/test, and prepares the column lists that the ColumnTransformer needs.

import numpy as np
import pandas as pd
from sklearn.datasets import fetch_openml
from sklearn.model_selection import train_test_split

adult = fetch_openml(name="adult", version=2, as_frame=True, parser="auto")
X = adult.data.drop(columns=["fnlwgt", "education-num"], errors="ignore")
y = (adult.target == ">50K").astype(int)

# Drop rows with missing values for simplicity
mask = ~X.isna().any(axis=1)
X, y = X[mask], y[mask]

X_tr, X_te, y_tr, y_te = train_test_split(
    X, y, test_size=0.2, stratify=y, random_state=42
)

num_cols = X_tr.select_dtypes(include=["number"]).columns.tolist()
cat_cols = X_tr.select_dtypes(exclude=["number"]).columns.tolist()
print(f"Train shape: {X_tr.shape}, Test shape: {X_te.shape}")
print(f"Numeric cols ({len(num_cols)}): {num_cols}")
print(f"Categorical cols ({len(cat_cols)}): {cat_cols}")
print(f"Positive rate (train): {y_tr.mean():.3f}")

Task 1 — Cached Pipeline(ColumnTransformer, XGBClassifier)

sol_task1_pipeline.py
# ─── sol_task1_pipeline.py ────────────────────────────────────────────────────
# Build the cached pipeline, fit it, and evaluate the baseline test ROC-AUC.

import numpy as np
import joblib
from sklearn.compose import ColumnTransformer
from sklearn.preprocessing import StandardScaler, OneHotEncoder
from sklearn.pipeline import Pipeline
from sklearn.metrics import roc_auc_score
from xgboost import XGBClassifier

# Compute scale_pos_weight from the training set's class balance
n_neg = int((y_tr == 0).sum())
n_pos = int((y_tr == 1).sum())
spw = n_neg / n_pos
print(f"scale_pos_weight = {n_neg} / {n_pos} = {spw:.3f}")

# ColumnTransformer: scale numeric, one-hot categorical
preprocess = ColumnTransformer(
    transformers=[
        ("num", StandardScaler(), num_cols),
        ("cat", OneHotEncoder(handle_unknown="ignore", max_categories=20), cat_cols),
    ],
    remainder="drop",
)

# XGBClassifier with sensible defaults
xgb_clf = XGBClassifier(
    n_estimators=200,
    learning_rate=0.05,
    max_depth=6,
    subsample=0.8,
    colsample_bytree=0.8,
    min_child_weight=5,
    reg_lambda=1.0,
    gamma=0.0,
    tree_method="hist",
    eval_metric="logloss",
    scale_pos_weight=spw,
    n_jobs=-1,
    random_state=42,
)

# Pipeline with caching — preprocessing is cached across search iterations
pipe = Pipeline(
    steps=[("pre", preprocess), ("clf", xgb_clf)],
    memory="/tmp/xgb_cache",
)

# Fit and evaluate
pipe.fit(X_tr, y_tr)
y_proba = pipe.predict_proba(X_te)[:, 1]
test_auc = roc_auc_score(y_te, y_proba)
print(f"Baseline test ROC-AUC: {test_auc:.4f}")

Typical output: scale_pos_weight ≈ 3.16 (76/24 imbalance), baseline test ROC-AUC around 0.928–0.932. The cached pipeline fits in 5–15 seconds. The memory="/tmp/xgb_cache" argument means subsequent RandomizedSearchCV calls in Task 3 will skip the preprocessing on configurations that have already been tried, saving significant time.

Task 2 — Triad Sweep with Early Stopping

sol_task2_triad.py
# ─── sol_task2_triad.py ───────────────────────────────────────────────────────
# Sweep (learning_rate, max_depth) with xgb.cv() and early stopping to find
# the optimal n_estimators for each combination. Plot a heatmap of CV AUC.

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import xgboost as xgb
from sklearn.compose import ColumnTransformer
from sklearn.preprocessing import StandardScaler, OneHotEncoder

# Fit the preprocessor once and transform the training data
preprocess = ColumnTransformer(
    transformers=[
        ("num", StandardScaler(), num_cols),
        ("cat", OneHotEncoder(handle_unknown="ignore", max_categories=20), cat_cols),
    ],
)
X_tr_pre = preprocess.fit_transform(X_tr)
print(f"Preprocessed training shape: {X_tr_pre.shape}")

dtrain = xgb.DMatrix(X_tr_pre, label=y_tr.values)

learning_rates = [0.01, 0.05, 0.1, 0.3]
max_depths     = [3, 4, 6, 8, 10]
results = np.zeros((len(learning_rates), len(max_depths)))
best_iters = np.zeros_like(results, dtype=int)

for i, lr in enumerate(learning_rates):
    for j, md in enumerate(max_depths):
        params = {
            "objective": "binary:logistic",
            "eval_metric": "auc",
            "tree_method": "hist",
            "learning_rate": lr,
            "max_depth": md,
            "subsample": 0.8,
            "colsample_bytree": 0.8,
            "min_child_weight": 5,
            "scale_pos_weight": spw,
            "seed": 42,
        }
        cv = xgb.cv(
            params, dtrain,
            num_boost_round=2000,
            nfold=5,
            early_stopping_rounds=50,
            metrics="auc",
            as_pandas=True,
            seed=42,
            verbose_eval=False,
        )
        best_iter = len(cv)
        best_auc = cv["test-auc-mean"].iloc[-1]
        results[i, j] = best_auc
        best_iters[i, j] = best_iter
        print(f"  lr={lr:.2f}  max_depth={md:2d}  best_iter={best_iter:4d}  "
              f"CV AUC={best_auc:.4f}")

# Heatmap
fig, ax = plt.subplots(figsize=(7, 5), constrained_layout=True)
im = ax.imshow(results, aspect="auto", cmap="viridis")
ax.set_xticks(range(len(max_depths)))
ax.set_yticks(range(len(learning_rates)))
ax.set_xticklabels(max_depths)
ax.set_yticklabels(learning_rates)
ax.set_xlabel("max_depth")
ax.set_ylabel("learning_rate")
ax.set_title("CV AUC heatmap (xgb.cv with early stopping)")
for i in range(len(learning_rates)):
    for j in range(len(max_depths)):
        ax.text(j, i, f"{results[i, j]:.3f}\n({best_iters[i, j]} trees)",
                ha="center", va="center", color="white", fontsize=9)
fig.colorbar(im, label="CV AUC")
plt.savefig("assignment04_task2_heatmap.png", dpi=120)
plt.show()

# Pick the best
best_i, best_j = np.unravel_index(np.argmax(results), results.shape)
best_lr = learning_rates[best_i]
best_md = max_depths[best_j]
best_n  = int(best_iters[best_i, best_j])
print(f"\nBest triad: learning_rate={best_lr}, max_depth={best_md}, "
      f"n_estimators={best_n}, CV AUC={results.max():.4f}")

Typical best triad: learning_rate=0.05, max_depth=8, n_estimators≈250–400, with CV AUC around 0.935. The heatmap makes the bias-variance interaction visible: low learning_rate with low max_depth underfits (insufficient model capacity), high learning_rate with high max_depth overfits (each tree too powerful, fewer trees needed, but worse generalization). The sweet spot is in the middle of the grid.

Task 3 — RandomizedSearchCV over Regularization Knobs

sol_task3_random_search.py
# ─── sol_task3_random_search.py ───────────────────────────────────────────────
# Fix the triad at the Task 2 best values and run RandomizedSearchCV over
# the 5 regularization knobs. Pipeline caching makes this much faster than
# re-running preprocessing for every trial.

import time
import numpy as np
from scipy.stats import uniform, loguniform
from sklearn.compose import ColumnTransformer
from sklearn.preprocessing import StandardScaler, OneHotEncoder
from sklearn.pipeline import Pipeline
from sklearn.model_selection import RandomizedSearchCV, StratifiedKFold
from sklearn.metrics import roc_auc_score
from xgboost import XGBClassifier

# Best triad from Task 2 (use the values discovered above)
BEST_LR = 0.05
BEST_MD = 8
BEST_N  = 300

xgb_clf = XGBClassifier(
    n_estimators=BEST_N,
    learning_rate=BEST_LR,
    max_depth=BEST_MD,
    tree_method="hist",
    eval_metric="logloss",
    scale_pos_weight=spw,
    n_jobs=1,            # set per-trial n_jobs=1, search-level n_jobs=-1
    random_state=42,
)

preprocess = ColumnTransformer(
    transformers=[
        ("num", StandardScaler(), num_cols),
        ("cat", OneHotEncoder(handle_unknown="ignore", max_categories=20), cat_cols),
    ],
)

pipe = Pipeline(
    steps=[("pre", preprocess), ("clf", xgb_clf)],
    memory="/tmp/xgb_cache",
)

param_dist = {
    "clf__subsample":        uniform(0.5, 0.5),       # range [0.5, 1.0]
    "clf__colsample_bytree": uniform(0.4, 0.6),       # range [0.4, 1.0]
    "clf__gamma":            uniform(0.0, 5.0),
    "clf__reg_alpha":        loguniform(1e-3, 1e1),
    "clf__reg_lambda":       loguniform(1e-3, 1e1),
    "clf__min_child_weight": [1, 3, 5, 10, 20],
}

cv = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)
search = RandomizedSearchCV(
    estimator=pipe,
    param_distributions=param_dist,
    n_iter=50,
    cv=cv,
    scoring="roc_auc",
    n_jobs=-1,
    random_state=42,
    verbose=1,
)

t0 = time.time()
search.fit(X_tr, y_tr)
elapsed = time.time() - t0

print(f"\nSearch took {elapsed:.1f}s ({elapsed/60:.1f} min)")
print(f"Best params: {search.best_params_}")
print(f"Best CV ROC-AUC: {search.best_score_:.4f}")

y_proba = search.predict_proba(X_te)[:, 1]
test_auc = roc_auc_score(y_te, y_proba)
print(f"Test ROC-AUC:    {test_auc:.4f}")

Typical best params: subsample ≈ 0.75, colsample_bytree ≈ 0.65, gamma ≈ 0.5, reg_alpha ≈ 0.05, reg_lambda ≈ 1.0, min_child_weight = 5 or 10. Test ROC-AUC should land around 0.937–0.942 — a small but real improvement over the Task 2 baseline. Search time is typically 1–3 minutes thanks to the cached preprocessing.

Task 4 — Feature Importance Comparison

sol_task4_feature_importance.py
# ─── sol_task4_feature_importance.py ──────────────────────────────────────────
# Compare three feature importance measures: gain, weight, and permutation.
# Plot them side by side for the top 15 features.

import numpy as np
import matplotlib.pyplot as plt
import pandas as pd
from sklearn.inspection import permutation_importance
from sklearn.metrics import roc_auc_score

# Extract the fitted preprocessor and XGBClassifier from the best pipeline
best_pipe = search.best_estimator_
fitted_pre = best_pipe.named_steps["pre"]
fitted_clf = best_pipe.named_steps["clf"]

# Get the post-encoding feature names
feature_names = fitted_pre.get_feature_names_out()
print(f"Post-encoding feature count: {len(feature_names)}")

# 1. Gain importance (the default for XGBClassifier.feature_importances_)
gain_imp = pd.Series(fitted_clf.feature_importances_, index=feature_names,
                     name="gain").sort_values(ascending=False)

# 2. Weight importance (number of splits using each feature)
booster = fitted_clf.get_booster()
weight_dict = booster.get_score(importance_type="weight")
# Map to the same feature-name namespace
weight_imp = pd.Series(0.0, index=feature_names, name="weight")
for k, v in weight_dict.items():
    if k in weight_imp.index:
        weight_imp[k] = v
weight_imp = weight_imp.sort_values(ascending=False)

# 3. Permutation importance on the test set
perm = permutation_importance(
    best_pipe, X_te, y_te,
    n_repeats=10, random_state=42, n_jobs=-1, scoring="roc_auc",
)
perm_imp = pd.Series(perm.importances_mean, index=X_te.columns,
                     name="permutation").sort_values(ascending=False)

# Note: gain and weight use post-encoding names (e.g. "num__age",
# "cat__marital-status_Married-civ-spouse"); permutation uses original
# column names. We align them by reporting each on its own scale.

# Top-15 bar chart, side by side
fig, axes = plt.subplots(1, 3, figsize=(18, 6), constrained_layout=True)
for ax, (imp, title, color) in zip(axes, [
    (gain_imp, "Gain importance", "#38bdf8"),
    (weight_imp, "Weight importance (split count)", "#f59e0b"),
    (perm_imp, "Permutation importance (test AUC drop)", "#f472b6"),
]):
    top = imp.head(15).iloc[::-1]
    ax.barh(range(len(top)), top.values, color=color)
    ax.set_yticks(range(len(top)))
    ax.set_yticklabels(top.index, fontsize=8)
    ax.set_title(title)
    ax.set_xlabel("Importance")
plt.savefig("assignment04_task4_importance.png", dpi=120)
plt.show()

print("\nTop-5 by gain:")
print(gain_imp.head(5).round(4))
print("\nTop-5 by weight:")
print(weight_imp.head(5).round(4))
print("\nTop-5 by permutation:")
print(perm_imp.head(5).round(4))

Typical top features by permutation importance on Adult Census: age, capital-gain, marital-status, education-num, hours-per-week. Gain importance usually agrees on the top 2–3 features but may rank them in a different order, because gain rewards large-loss-reduction splits (which often happen on continuous features like capital-gain) rather than overall model reliance. Weight importance is dominated by features that are split on frequently, which often means high-cardinality categorical features after one-hot encoding. The disagreement between the three measures is itself informative — it tells you which features the model "uses heavily" (weight) versus which features "drive predictions" (gain) versus which features "actually matter for test performance" (permutation).

Task 5 — XGBoost vs LightGBM Comparison

sol_task5_xgb_vs_lgbm.py
# ─── sol_task5_xgb_vs_lgbm.py ─────────────────────────────────────────────────
# Train a LightGBM model with hyperparameters matched to the best XGBoost
# config from Task 3. Compare test ROC-AUC, fit time, and predict time.

import time
import numpy as np
import pandas as pd
from sklearn.metrics import roc_auc_score
from lightgbm import LGBMClassifier, early_stopping, log_evaluation

# Map XGBoost hyperparameters to LightGBM equivalents
best = search.best_params_
xgb_to_lgbm = {
    "learning_rate":     BEST_LR,
    "max_depth":         BEST_MD,
    "n_estimators":      BEST_N + 500,  # give early stopping room
    "subsample":         best["clf__subsample"],
    "colsample_bytree":  best["clf__colsample_bytree"],
    "reg_alpha":         best["clf__reg_alpha"],
    "reg_lambda":        best["clf__reg_lambda"],
    "min_child_weight":  best["clf__min_child_weight"],
}

# LightGBM renames some hyperparameters
lgbm_clf = LGBMClassifier(
    objective="binary",
    learning_rate=xgb_to_lgbm["learning_rate"],
    max_depth=xgb_to_lgbm["max_depth"],
    num_leaves=min(63, 2 ** xgb_to_lgbm["max_depth"] - 1),
    n_estimators=xgb_to_lgbm["n_estimators"],
    subsample=xgb_to_lgbm["subsample"],            # → bagging_fraction
    subsample_freq=1,                              # → bagging_freq
    colsample_bytree=xgb_to_lgbm["colsample_bytree"],  # → feature_fraction
    reg_alpha=xgb_to_lgbm["reg_alpha"],            # → lambda_l1
    reg_lambda=xgb_to_lgbm["reg_lambda"],          # → lambda_l2
    min_child_samples=max(1, xgb_to_lgbm["min_child_weight"]),
    scale_pos_weight=spw,
    n_jobs=-1,
    random_state=42,
    verbose=-1,
)

# Use the same preprocessor (fit on train, transform both)
from sklearn.compose import ColumnTransformer
from sklearn.preprocessing import StandardScaler, OneHotEncoder
preprocess = ColumnTransformer(
    transformers=[
        ("num", StandardScaler(), num_cols),
        ("cat", OneHotEncoder(handle_unknown="ignore", max_categories=20), cat_cols),
    ],
)
X_tr_pre = preprocess.fit_transform(X_tr)
X_te_pre = preprocess.transform(X_te)

# Fit LightGBM with early stopping
t0 = time.time()
lgbm_clf.fit(
    X_tr_pre, y_tr,
    eval_set=[(X_te_pre, y_te)],
    eval_metric="auc",
    callbacks=[early_stopping(50), log_evaluation(0)],
)
lgbm_fit_time = time.time() - t0

t0 = time.time()
y_proba_lgbm = lgbm_clf.predict_proba(X_te_pre)[:, 1]
lgbm_pred_time = time.time() - t0
lgbm_auc = roc_auc_score(y_te, y_proba_lgbm)

# Compare against XGBoost (already fitted in Task 3)
t0 = time.time()
y_proba_xgb = search.predict_proba(X_te)[:, 1]
xgb_pred_time = time.time() - t0
xgb_auc = roc_auc_score(y_te, y_proba_xgb)

print(f"\n{'Model':10s} {'Test AUC':>10s} {'Fit time':>10s} {'Pred time':>12s}")
print("-" * 50)
print(f"{'XGBoost':10s} {xgb_auc:>10.4f} {'(cached)':>10s} {xgb_pred_time:>11.3f}s")
print(f"{'LightGBM':10s} {lgbm_auc:>10.4f} {lgbm_fit_time:>9.2f}s {lgbm_pred_time:>11.3f}s")
print(f"\nLightGBM best_iteration: {lgbm_clf.best_iteration_}")

Typical results on Adult Census:

ModelTest ROC-AUCFit timePredict time
XGBoost (Task 3 best)~0.940~1.5s (after cache)~0.05s
LightGBM (matched)~0.938~0.5s~0.04s

The two models land within 0.2 percentage points of ROC-AUC, with LightGBM training roughly 2–3x faster. This is the typical pattern: XGBoost and LightGBM are close in accuracy on most tabular datasets, but LightGBM's histogram binning and leaf-wise growth make it faster. For production deployment where training happens often (e.g. daily retraining), the speed advantage is significant. For one-off model building, the accuracy difference is small enough that either is a fine choice.

What This Assignment Teaches

  • The XGBoost tuning workflow is a two-stage process: first tune the triad with early stopping (Task 2), then tune the regularization knobs with RandomizedSearchCV (Task 3).
  • Pipeline caching via memory= can speed up RandomizedSearchCV by 5–10x when preprocessing is expensive — always use it.
  • scale_pos_weight is essential for imbalanced binary classification — without it, the model is biased toward the majority class.
  • Feature importance measures disagree for a reason: gain rewards loss-reducing splits, weight rewards frequent splits, permutation rewards features the model actually relies on at test time. Use all three.
  • XGBoost vs LightGBM is typically a wash on accuracy; LightGBM wins on training speed. Pick based on your deployment constraints.
