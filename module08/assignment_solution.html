Below are reference solutions for every task in Assignment 08. The solutions use Fashion-MNIST for speed — each schedule trains in ~5–10 minutes on CPU. Use the same code with CIFAR-10 (swap the dataset and adjust the input channels from 1 to 3) and you will get qualitatively similar schedule comparisons at higher absolute accuracy.

Task 1 — Build the CNN

sol_task1_cnn.py
# ─── sol_task1_cnn.py ──────────────────────────────────────────────────────────
import torch
import torch.nn as nn
import torch.optim as optim
from torch.utils.data import DataLoader
from torchvision import datasets, transforms
import time

DEVICE = "cuda" if torch.cuda.is_available() else "cpu"

# ─── Data ─────────────────────────────────────────────────────────────────────
transform = transforms.Compose([
    transforms.ToTensor(),
    transforms.Normalize((0.2860,), (0.3530,)),   # Fashion-MNIST mean/std
])
train_ds = datasets.FashionMNIST("./data", train=True,  download=True, transform=transform)
test_ds  = datasets.FashionMNIST("./data", train=False, download=True, transform=transform)
train_loader = DataLoader(train_ds, batch_size=128, shuffle=True,  num_workers=2)
test_loader  = DataLoader(test_ds,  batch_size=256, shuffle=False, num_workers=2)

# ─── Model ────────────────────────────────────────────────────────────────────
class SmallCNN(nn.Module):
    def __init__(self):
        super().__init__()
        self.features = nn.Sequential(
            nn.Conv2d(1, 32, 3, padding=1), nn.BatchNorm2d(32), nn.ReLU(),
            nn.MaxPool2d(2),                                   # 28 -> 14
            nn.Conv2d(32, 64, 3, padding=1), nn.BatchNorm2d(64), nn.ReLU(),
            nn.MaxPool2d(2),                                   # 14 -> 7
        )
        self.classifier = nn.Sequential(
            nn.Flatten(),
            nn.Linear(64 * 7 * 7, 128), nn.ReLU(), nn.Dropout(0.3),
            nn.Linear(128, 10),
        )
    def forward(self, x):
        return self.classifier(self.features(x))

def train_one_epoch(model, loader, optimizer, criterion, device):
    model.train()
    total_loss, total_n = 0.0, 0
    for X, y in loader:
        X, y = X.to(device), y.to(device)
        optimizer.zero_grad()
        out = model(X)
        loss = criterion(out, y)
        loss.backward()
        optimizer.step()
        total_loss += loss.item() * len(y)
        total_n += len(y)
    return total_loss / total_n

@torch.no_grad()
def evaluate(model, loader, criterion, device):
    model.eval()
    total_loss, correct, total = 0.0, 0, 0
    for X, y in loader:
        X, y = X.to(device), y.to(device)
        out = model(X)
        total_loss += criterion(out, y).item() * len(y)
        correct  += (out.argmax(1) == y).sum().item()
        total    += len(y)
    return total_loss / total, correct / total

# ─── Sanity check ─────────────────────────────────────────────────────────────
torch.manual_seed(0)
model = SmallCNN().to(DEVICE)
opt = optim.Adam(model.parameters(), lr=1e-3)
crit = nn.CrossEntropyLoss()
loss = train_one_epoch(model, train_loader, opt, crit, DEVICE)
val_loss, val_acc = evaluate(model, test_loader, crit, DEVICE)
print(f"After 1 epoch: train_loss={loss:.4f}, val_loss={val_loss:.4f}, val_acc={val_acc:.4f}")

After 1 epoch with Adam at lr=1e-3, you should see val accuracy around 0.85 on Fashion-MNIST. If you see below 0.75, check (a) BatchNorm is present and in train mode during training, (b) CrossEntropyLoss is used (not NLLLoss with raw logits), (c) the data normalization constants match Fashion-MNIST's mean/std.

Task 2 — Train Under Five Schedules

sol_task2_schedules.py
# ─── sol_task2_schedules.py ────────────────────────────────────────────────────
import copy
EPOCHS = 20

def make_scheduler(name, optimizer, epochs=EPOCHS, steps_per_epoch=len(train_loader)):
    if name == "fixed":
        return None
    if name == "step":
        return optim.lr_scheduler.StepLR(optimizer, step_size=5, gamma=0.5)
    if name == "cosine":
        return optim.lr_scheduler.CosineAnnealingLR(optimizer, T_max=epochs, eta_min=1e-5)
    if name == "sgdr":
        return optim.lr_scheduler.CosineAnnealingWarmRestarts(
            optimizer, T_0=5, T_mult=1, eta_min=1e-5
        )
    if name == "onecycle":
        return optim.lr_scheduler.OneCycleLR(
            optimizer, max_lr=3e-3,
            total_steps=epochs * steps_per_epoch,
            pct_start=0.3, anneal_strategy="cos",
        )
    raise ValueError(name)

def train_with_schedule(schedule_name, epochs=EPOCHS, seed=0):
    torch.manual_seed(seed)
    model = SmallCNN().to(DEVICE)
    optimizer = optim.Adam(model.parameters(), lr=1e-3)
    scheduler = make_scheduler(schedule_name, optimizer, epochs)
    crit = nn.CrossEntropyLoss()

    val_accs, lr_track = [], []
    for ep in range(epochs):
        # OneCycleLR steps per-batch; others per-epoch.
        per_batch = schedule_name == "onecycle"
        if per_batch:
            model.train()
            for X, y in train_loader:
                X, y = X.to(DEVICE), y.to(DEVICE)
                optimizer.zero_grad()
                loss = crit(model(X), y)
                loss.backward()
                optimizer.step()
                scheduler.step()
        else:
            train_one_epoch(model, train_loader, optimizer, crit, DEVICE)
        val_loss, val_acc = evaluate(model, test_loader, crit, DEVICE)
        val_accs.append(val_acc)
        lr_track.append(optimizer.param_groups[0]["lr"])
        if not per_batch and scheduler is not None:
            scheduler.step()
    return val_accs, lr_track

# ─── Run all 5 schedules ──────────────────────────────────────────────────────
results = {}
for name in ["fixed", "step", "cosine", "sgdr", "onecycle"]:
    t0 = time.perf_counter()
    val_accs, lr_track = train_with_schedule(name, seed=0)
    dt = time.perf_counter() - t0
    results[name] = (val_accs, lr_track)
    print(f"{name:10s}  final_val_acc={val_accs[-1]:.4f}  ({dt:.1f}s)")

Typical output: fixed ends at ~0.890, step at ~0.900, cosine at ~0.910, sgdr at ~0.908, onecycle at ~0.912. The key thing to notice is the trajectory: onecycle rises fastest (it's at 0.88+ by epoch 5), cosine and sgdr rise steadily, step has visible plateaus at the step boundaries, fixed plateaus early and never refines.

The left panel should show 5 curves that separate visibly by epoch 5 and converge within ~2% of each other by the end. The right panel should show the 5 LR shapes from Figure 1 of the article. Task 4 — 3-Seed Stability Comparison

sol_task4_stability.py
# ─── sol_task4_stability.py ────────────────────────────────────────────────────
import numpy as np
import pandas as pd

rows = []
THRESH = 0.88
for name in ["fixed", "step", "cosine", "sgdr", "onecycle"]:
    finals, peaks, epochs_to_thresh = [], [], []
    for seed in [0, 1, 2]:
        val_accs, _ = train_with_schedule(name, seed=seed)
        finals.append(val_accs[-1])
        peaks.append(max(val_accs))
        crossed = next((i for i, v in enumerate(val_accs) if v >= THRESH), None)
        epochs_to_thresh.append(crossed if crossed is not None else -1)
    rows.append({
        "schedule":      name,
        "mean_final":    np.mean(finals),
        "std_final":     np.std(finals),
        "mean_peak":     np.mean(peaks),
        "mean_epochs_to_88%": float(np.mean([e for e in epochs_to_thresh if e >= 0])
                                    if any(e >= 0 for e in epochs_to_thresh) else -1),
    })
df = pd.DataFrame(rows)
print(df.to_string(index=False))

Three observations: (1) One Cycle is fastest — it crosses 88% by epoch 4 because the rising-LR phase lets the optimizer cover ground quickly. (2) Cosine and One Cycle are tied for best final — the difference is within noise. (3) Fixed LR has the highest variance — without annealing, the model ends up at a sharper minimum whose exact location depends on the seed. Schedules that anneal smoothly to near-zero (cosine, One Cycle) consistently find flatter minima that generalize more reliably.

Task 5 (Bonus) — Warmup Duration Sweep

sol_task5_warmup_sweep.py
# ─── sol_task5_warmup_sweep.py ─────────────────────────────────────────────────
import math
from torch.optim.lr_scheduler import LambdaLR

def get_warmup_cosine(optimizer, warmup_epochs, total_epochs, base_lr=1e-3, min_lr_ratio=1e-2):
    def lr_lambda(epoch):
        if epoch < warmup_epochs:
            return float(epoch) / float(max(1, warmup_epochs))
        progress = float(epoch - warmup_epochs) / float(max(1, total_epochs - warmup_epochs))
        cos = 0.5 * (1.0 + math.cos(math.pi * progress))
        return min_lr_ratio + (1.0 - min_lr_ratio) * cos
    return LambdaLR(optimizer, lr_lambda=lr_lambda)

def train_warmup_cosine(warmup_epochs, epochs=EPOCHS, seed=0):
    torch.manual_seed(seed)
    model = SmallCNN().to(DEVICE)
    optimizer = optim.Adam(model.parameters(), lr=1e-3)
    scheduler = get_warmup_cosine(optimizer, warmup_epochs, epochs)
    crit = nn.CrossEntropyLoss()
    val_accs = []
    for ep in range(epochs):
        train_one_epoch(model, train_loader, optimizer, crit, DEVICE)
        _, va = evaluate(model, test_loader, crit, DEVICE)
        val_accs.append(va)
        scheduler.step()
    return val_accs

# ─── Sweep warmup durations ──────────────────────────────────────────────────
plt.figure(figsize=(9, 5), constrained_layout=True)
for w in [0, 1, 5, 10]:
    va = train_warmup_cosine(w, seed=0)
    plt.plot(va, "-o", label=f"warmup={w} (final={va[-1]:.3f})", markersize=4)
plt.xlabel("Epoch"); plt.ylabel("Val accuracy")
plt.title("Warmup duration sweep (cosine decay, 20 epochs)")
plt.legend(loc="lower right", fontsize=10); plt.grid(alpha=0.2)
plt.savefig("assignment08_warmup_sweep.png", dpi=120)
plt.show()

Explanation

Typical result: warmup of 0–1 epochs gives the best final accuracy on Fashion-MNIST; warmup of 5 epochs is slightly worse (the model spends 5 epochs making slow progress at low LR); warmup of 10 epochs is clearly worse (half of training is wasted on the warmup ramp). This makes sense because Fashion-MNIST is a relatively easy dataset where Adam's variance estimate converges quickly — there is no need for the long warmup that Transformers require.

The general rule: warmup is most valuable when (a) the model is unstable at the start (Transformers, large batch sizes, RL), (b) the optimizer's variance estimate needs time to converge (Adam at high LR, fresh models with random initialization). On stable problems with well-conditioned models (CNNs on Fashion-MNIST, ResNet on ImageNet with BatchNorm), warmup of 0–1 epoch is fine; long warmup actively hurts because it delays the productive training phase. This is why BERT uses 10k warmup steps out of 1M (1%) but ResNet uses no warmup at all — different model families have different stability properties.

🎯 What This Assignment Teaches

  • The schedule shape, not just the average LR, determines final accuracy — cosine annealing and One Cycle reach 1–3% higher accuracy than fixed LR in the same number of epochs.
  • One Cycle is fastest to converge: the rising-LR phase lets the optimizer make large early progress, and the cosine anneal refines to a good minimum.
  • Cosine annealing is the safe default for CNNs — single parameter (T_max), no LR finder needed, near-optimal across most tasks.
  • Schedule choice has lower variance across seeds than fixed LR — annealing to near-zero finds flatter minima that generalize more reliably.
  • Warmup duration should match the model's stability needs: 0–1 epoch for CNNs with BatchNorm, 1–10% of total for Transformers.
  • The qualitative ranking (One Cycle ≈ cosine > step > fixed) is robust across datasets and seeds; the absolute numbers vary, but the ordering is reliable.
