Task 1 — Build the Pipeline

snippet.py
# ─── sol_task1_pipeline.py ─────────────────────────────────────────────────────
import tempfile
from pathlib import Path
from joblib import Memory
from sklearn.compose import ColumnTransformer, make_column_selector
from sklearn.datasets import fetch_openml
from sklearn.ensemble import RandomForestClassifier
from sklearn.impute import SimpleImputer
from sklearn.pipeline import Pipeline
from sklearn.preprocessing import OneHotEncoder

X, y = fetch_openml(name="adult", version=2, as_frame=True, return_X_y=True)

cat_cols = X.select_dtypes(include=["category", "object"]).columns.tolist()
num_cols = X.select_dtypes(exclude=["category", "object"]).columns.tolist()

cache_dir = Path(tempfile.gettempdir()) / "hypertune_adult_cache"
cache_dir.mkdir(exist_ok=True)
memory = Memory(location=str(cache_dir), verbose=0)

preprocess = ColumnTransformer([
    ("cat", Pipeline([
        ("imputer", SimpleImputer(strategy="most_frequent")),
        ("onehot", OneHotEncoder(handle_unknown="ignore", min_frequency=0.01)),
    ]), cat_cols),
    ("num", SimpleImputer(strategy="median"), num_cols),
])

pipe = Pipeline([
    ("preprocess", preprocess),
    ("rf", RandomForestClassifier(n_estimators=300, n_jobs=1, random_state=42)),
], memory=memory)
print(pipe)

Task 2 — Coarse Grid Search

snippet.py
# ─── sol_task2_coarse.py ───────────────────────────────────────────────────────
import pandas as pd
from sklearn.model_selection import GridSearchCV, StratifiedKFold

coarse_grid = {
    "rf__max_depth":        [4, 8, 12, 16],
    "rf__min_samples_leaf": [1, 4, 16, 64],
    "rf__max_features":     ["sqrt", "log2", 0.5],
}
cv = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)
coarse = GridSearchCV(pipe, param_grid=coarse_grid, cv=cv,
                      scoring="accuracy", n_jobs=-1, return_train_score=True)
coarse.fit(X, y)
print(f"Coarse best: {coarse.best_params_}  acc = {coarse.best_score_:.4f}")

df = pd.DataFrame(coarse.cv_results_)
top5 = df.nlargest(5, "mean_test_score")[
    ["params", "mean_test_score", "std_test_score", "mean_fit_time"]
]
print(top5.to_string(index=False))

Typical output: best around max_depth=16, min_samples_leaf=4, max_features='sqrt' with CV accuracy ~0.865.

Task 3 — Fine Grid

snippet.py
# ─── sol_task3_fine.py ─────────────────────────────────────────────────────────
import numpy as np
best = coarse.best_params_
fine_grid = {
    "rf__max_depth":        list(range(best["rf__max_depth"] - 2, best["rf__max_depth"] + 3)),
    "rf__min_samples_leaf": [max(1, best["rf__min_samples_leaf"] // 2),
                             best["rf__min_samples_leaf"],
                             best["rf__min_samples_leaf"] * 2],
    "rf__max_features":     [best["rf__max_features"]],
}
fine = GridSearchCV(pipe, param_grid=fine_grid, cv=cv,
                    scoring="accuracy", n_jobs=-1, return_train_score=True)
fine.fit(X, y)
print(f"Fine best: {fine.best_params_}  acc = {fine.best_score_:.4f}")
print(f"Coarse best was: {coarse.best_score_:.4f}")

Task 4 — Heatmap

snippet.py
# ─── sol_task4_heatmap.py ──────────────────────────────────────────────────────
import matplotlib.pyplot as plt
import seaborn as sns

df = pd.DataFrame(fine.cv_results_)
pivot = df.pivot_table(index="param_rf__max_depth",
                       columns="param_rf__min_samples_leaf",
                       values="mean_test_score")
plt.figure(figsize=(8, 5), constrained_layout=True)
sns.heatmap(pivot, annot=True, fmt=".4f", cmap="viridis",
            cbar_kws={"label": "CV accuracy"})
plt.title("Fine grid: RandomForest on Adult Census")
plt.savefig("assignment02_heatmap.png", dpi=120)
plt.show()

Task 5 — Stability Analysis

snippet.py
# ─── sol_task5_stability.py ────────────────────────────────────────────────────
import numpy as np
import matplotlib.pyplot as plt

df = pd.DataFrame(fine.cv_results_)
df["lower"] = df["mean_test_score"] - df["std_test_score"]
top8 = df.nlargest(8, "mean_test_score").reset_index(drop=True)

split_cols = [c for c in df.columns if c.startswith("split") and c.endswith("test_score")]
fig, ax = plt.subplots(figsize=(10, 5), constrained_layout=True)
ax.boxplot([top8[c].values for c in split_cols], labels=[f"#{i+1}" for i in range(len(top8))])
ax.set_ylabel("Per-fold accuracy")
ax.set_title("Stability of top 8 fine-grid configurations")
plt.savefig("assignment02_stability.png", dpi=120)
plt.show()

# Rank by (mean - std) vs by mean
df["rank_by_mean"] = df["mean_test_score"].rank(ascending=False).astype(int)
df["rank_by_lower"] = df["lower"].rank(ascending=False).astype(int)
compare = df.nsmallest(8, "rank_by_mean")[
    ["params", "mean_test_score", "std_test_score", "lower",
     "rank_by_mean", "rank_by_lower"]
]
print(compare.to_string(index=False))

Often the top config by mean is rank 3 or 4 by stability. This is the kind of insight that production tuning should consider — picking the slightly-lower-mean but much-more-stable configuration is usually the right call for deployment.

🎯 What This Assignment Teaches

  • Coarse-to-fine is dramatically cheaper than one big grid and usually finds a better optimum.
  • cv_results_ is your primary analysis tool — convert to pandas and slice.
  • Heatmaps make 2D interactions visible; box plots make stability visible.
  • Ranking by (mean - std) often changes the top pick compared to ranking by mean alone.
  • joblib.Memory makes a real difference for pipelines with expensive transformers.
